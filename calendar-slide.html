const ICS_URL = 'https://calendar.google.com/calendar/ical/2be647771ecee6ed081baac45985a30286da39d8debbde7eb3a7e23af1110303%40group.calendar.google.com/private-5b40b5eb03345f30795e55e26864bf95/basic.ics'; const OUTPUT_TZ = 'Europe/Oslo'; function doGet() { const ics = UrlFetchApp.fetch(ICS_URL, { muteHttpExceptions: false, followRedirects: true }).getContentText(); const now = new Date(); const events = parseIcsEvents_(ics) .filter(ev => ev.status !== 'CANCELLED') .filter(ev => ev.beginDate) .filter(ev => { const end = ev.endDate || ev.beginDate; return end >= now; }) .sort((a, b) => a.beginDate - b.beginDate) .slice(0, 120) .map(ev => ({ title: ev.title || '', begin: formatDateTime_(ev.beginDate), end: formatDateTime_(ev.endDate), place: ev.location || '' })); const payload = { updated: formatDateTime_(new Date()), title: 'Dette kan du oppleve i Åmot kulturhus', events }; return ContentService .createTextOutput(JSON.stringify(payload)) .setMimeType(ContentService.MimeType.JSON); } function parseIcsEvents_(icsText) { const text = unfoldIcs_(icsText); const blocks = text .split('BEGIN:VEVENT') .slice(1) .map(part => part.split('END:VEVENT')[0]); return blocks .map(parseEventBlock_) .filter(Boolean); } function unfoldIcs_(text) { return String(text).replace(/\r?\n[ \t]/g, ''); } function parseEventBlock_(block) { const lines = String(block) .split(/\r?\n/) .map(s => s.trim()) .filter(Boolean); const fields = {}; for (const line of lines) { const idx = line.indexOf(':'); if (idx < 0) continue; const left = line.slice(0, idx); const value = line.slice(idx + 1); const name = left.split(';')[0].toUpperCase(); if (!fields[name]) fields[name] = []; fields[name].push({ rawKey: left, value }); } const status = getValue_(fields, 'STATUS'); const summary = getValue_(fields, 'SUMMARY'); const location = getValue_(fields, 'LOCATION'); const dtStart = getEntry_(fields, 'DTSTART'); const dtEnd = getEntry_(fields, 'DTEND'); const beginDate = parseIcsDate_(dtStart); let endDate = parseIcsDate_(dtEnd); if (!beginDate) return null; if (!endDate) { endDate = new Date(beginDate.getTime()); } return { status: status || '', title: unescapeIcsText_(summary || ''), location: unescapeIcsText_(location || ''), beginDate, endDate }; } function getEntry_(fields, key) { return fields[key] && fields[key].length ? fields[key][0] : null; } function getValue_(fields, key) { const entry = getEntry_(fields, key); return entry ? entry.value : ''; } function parseIcsDate_(entry) { if (!entry || !entry.value) return null; const rawKey = entry.rawKey || ''; const value = String(entry.value).trim(); if (/VALUE=DATE/i.test(rawKey) || /^\d{8}$/.test(value)) { const year = Number(value.slice(0, 4)); const month = Number(value.slice(4, 6)); const day = Number(value.slice(6, 8)); const d = new Date(year, month - 1, day, 0, 0, 0); return Number.isNaN(d.getTime()) ? null : d; } if (/^\d{8}T\d{6}Z$/.test(value)) { const year = Number(value.slice(0, 4)); const month = Number(value.slice(4, 6)); const day = Number(value.slice(6, 8)); const hour = Number(value.slice(9, 11)); const minute = Number(value.slice(11, 13)); const second = Number(value.slice(13, 15)); return new Date(Date.UTC(year, month - 1, day, hour, minute, second)); } if (/^\d{8}T\d{6}$/.test(value)) { const year = Number(value.slice(0, 4)); const month = Number(value.slice(4, 6)); const day = Number(value.slice(6, 8)); const hour = Number(value.slice(9, 11)); const minute = Number(value.slice(11, 13)); const second = Number(value.slice(13, 15)); const d = new Date(year, month - 1, day, hour, minute, second); return Number.isNaN(d.getTime()) ? null : d; } return null; } function formatDateTime_(dateObj) { if (!dateObj) return ''; return Utilities.formatDate(dateObj, OUTPUT_TZ, 'yyyy-MM-dd HH:mm:ss'); } function unescapeIcsText_(text) { return String(text) .replace(/\\\\/g, '\\') .replace(/\\n/gi, '\n') .replace(/\\,/g, ',') .replace(/\\;/g, ';'); }
